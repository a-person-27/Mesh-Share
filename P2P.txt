import os
import shutil
import socket
import threading
import json
import random
import tkinter as tk
from tkinter import ttk, filedialog, messagebox
from zeroconf import Zeroconf, ServiceInfo, ServiceBrowser
from tkinterdnd2 import TkinterDnD, DND_FILES

SERVICE_TYPE = "_meshshae._tcp.local."
SHARED_FOLDER = "./shared"
BUFFER_SIZE = 65536


def get_local_ip():
    s = socket.socket(socket.AF_INET, socket.SOCK_DGRAM)
    s.connect(("8.8.8.8", 80))
    ip = s.getsockname()[0]
    s.close()
    return ip


def format_size(size):
    for unit in ['B', 'KB', 'MB', 'GB']:
        if size < 1024:
            return f"{size:.1f} {unit}"
        size /= 1024
    return f"{size:.1f} TB"


class PeerListener:
    def __init__(self):
        self.peers = {}

    def remove_service(self, zeroconf, type, name):
        self.peers.pop(name, None)

    def add_service(self, zeroconf, type, name):
        info = zeroconf.get_service_info(type, name)
        if info:
            addr = socket.inet_ntoa(info.addresses[0])
            self.peers[name] = (addr, info.port)


def handle_client(conn):
    try:
        request = conn.recv(1024).decode()

        if request == "LIST":
            files = []
            for f in os.listdir(SHARED_FOLDER):
                path = os.path.join(SHARED_FOLDER, f)
                if os.path.isfile(path):
                    files.append({"name": f, "size": os.path.getsize(path)})
            conn.send(json.dumps(files).encode())

        elif request.startswith("GET "):
            filename = request[4:]
            path = os.path.join(SHARED_FOLDER, filename)

            if os.path.exists(path):
                conn.send(b"OK")
                size = os.path.getsize(path)
                conn.send(str(size).encode().ljust(16))
                with open(path, "rb") as f:
                    while chunk := f.read(BUFFER_SIZE):
                        conn.sendall(chunk)
            else:
                conn.send(b"NOTFOUND")
    finally:
        conn.close()


def start_server(port):
    server = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
    server.bind(("0.0.0.0", port))
    server.listen()
    while True:
        conn, _ = server.accept()
        threading.Thread(target=handle_client, args=(conn,), daemon=True).start()


class MeshShaeApp:

    def __init__(self, root):
        self.root = root
        self.root.title("Mesh-Shae")
        self.root.geometry("1000x550")

        os.makedirs(SHARED_FOLDER, exist_ok=True)

        self.port = random.randint(50000, 51000)
        self.ip = get_local_ip()
        self.hostname = socket.gethostname()

        self.zeroconf = Zeroconf()
        self.listener = PeerListener()
        ServiceBrowser(self.zeroconf, SERVICE_TYPE, self.listener)

        info = ServiceInfo(
            SERVICE_TYPE,
            f"{self.hostname}.{SERVICE_TYPE}",
            addresses=[socket.inet_aton(self.ip)],
            port=self.port,
        )

        self.zeroconf.register_service(info)
        threading.Thread(target=start_server, args=(self.port,), daemon=True).start()

        self.build_ui()
        self.refresh_local_files()
        self.auto_refresh()

    # ---------------- UI ----------------

    def build_ui(self):
        main = ttk.Frame(self.root, padding=10)
        main.pack(fill="both", expand=True)

        main.columnconfigure(0, weight=1)
        main.columnconfigure(1, weight=1)
        main.columnconfigure(2, weight=1)

        # Local shared files
        local_frame = ttk.LabelFrame(main, text="Your Shared Files")
        local_frame.grid(row=0, column=0, sticky="nsew", padx=5)

        self.local_tree = ttk.Treeview(local_frame, columns=("size",), show="headings")
        self.local_tree.heading("size", text="Size")
        self.local_tree.pack(fill="both", expand=True, padx=5, pady=5)

        btn_frame = ttk.Frame(local_frame)
        btn_frame.pack(pady=5)

        ttk.Button(btn_frame, text="Add Files", command=self.add_files).pack(side="left", padx=5)
        ttk.Button(btn_frame, text="Add Folder", command=self.add_folder).pack(side="left", padx=5)
        ttk.Button(btn_frame, text="Delete Selected", command=self.delete_selected).pack(side="left", padx=5)

        # Drag & Drop
        self.root.drop_target_register(DND_FILES)
        self.root.dnd_bind("<<Drop>>", self.handle_drop)

        # Peers
        peer_frame = ttk.LabelFrame(main, text="Peers")
        peer_frame.grid(row=0, column=1, sticky="nsew", padx=5)

        self.peer_list = tk.Listbox(peer_frame)
        self.peer_list.pack(fill="both", expand=True, padx=5, pady=5)
        self.peer_list.bind("<Double-Button-1>", lambda e: self.load_files())

        # Files
        file_frame = ttk.LabelFrame(main, text="Peer Files")
        file_frame.grid(row=0, column=2, sticky="nsew", padx=5)

        self.file_tree = ttk.Treeview(file_frame, columns=("size",), show="headings")
        self.file_tree.heading("size", text="Size")
        self.file_tree.pack(fill="both", expand=True, padx=5, pady=5)
        self.file_tree.bind("<Double-Button-1>", lambda e: self.download_selected())

        bottom = ttk.Frame(self.root)
        bottom.pack(fill="x")

        self.progress = ttk.Progressbar(bottom)
        self.progress.pack(side="left", fill="x", expand=True, padx=5, pady=5)

        self.status = ttk.Label(bottom, text="Ready")
        self.status.pack(side="right", padx=10)

    # ---------------- Local File Management ----------------

    def refresh_local_files(self):
        self.local_tree.delete(*self.local_tree.get_children())
        for f in os.listdir(SHARED_FOLDER):
            path = os.path.join(SHARED_FOLDER, f)
            if os.path.isfile(path):
                self.local_tree.insert("", "end", values=(f, format_size(os.path.getsize(path))))

    def add_files(self):
        files = filedialog.askopenfilenames()
        for file in files:
            shutil.copy(file, SHARED_FOLDER)
        self.refresh_local_files()

    def add_folder(self):
        folder = filedialog.askdirectory()
        if folder:
            for file in os.listdir(folder):
                full = os.path.join(folder, file)
                if os.path.isfile(full):
                    shutil.copy(full, SHARED_FOLDER)
        self.refresh_local_files()

    def delete_selected(self):
        selected = self.local_tree.selection()
        for item in selected:
            filename = self.local_tree.item(item)["values"][0]
            os.remove(os.path.join(SHARED_FOLDER, filename))
        self.refresh_local_files()

    def handle_drop(self, event):
        files = self.root.tk.splitlist(event.data)
        for file in files:
            if os.path.isfile(file):
                shutil.copy(file, SHARED_FOLDER)
        self.refresh_local_files()

    # ---------------- Peer Logic ----------------

    def auto_refresh(self):
        self.peer_list.delete(0, tk.END)
        for name in self.listener.peers:
            self.peer_list.insert(tk.END, name)
        self.root.after(3000, self.auto_refresh)

    def get_selected_peer(self):
        sel = self.peer_list.curselection()
        if not sel:
            return None
        name = self.peer_list.get(sel[0])
        return self.listener.peers.get(name)

    def load_files(self):
        peer = self.get_selected_peer()
        if not peer:
            return
        try:
            files = list_files(peer)
            self.file_tree.delete(*self.file_tree.get_children())
            for f in files:
                self.file_tree.insert("", "end", values=(f["name"], format_size(f["size"])))
        except:
            self.status.config(text="Failed to load files")

    def download_selected(self):
        peer = self.get_selected_peer()
        if not peer:
            return
        selected = self.file_tree.selection()
        if not selected:
            return
        filename = self.file_tree.item(selected[0])["values"][0]
        save_path = filedialog.asksaveasfilename(initialfile=filename)
        if not save_path:
            return

        def progress_callback(received, total):
            percent = (received / total) * 100
            self.progress["value"] = percent
            self.root.update_idletasks()

        def run():
            download_file(peer, filename, save_path, progress_callback)

        threading.Thread(target=run, daemon=True).start()


def list_files(peer):
    ip, port = peer
    s = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
    s.connect((ip, port))
    s.send(b"LIST")
    data = s.recv(65536)
    s.close()
    return json.loads(data.decode())


def download_file(peer, filename, save_path, progress_callback):
    ip, port = peer
    s = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
    s.connect((ip, port))
    s.send(f"GET {filename}".encode())
    if s.recv(8) != b"OK":
        s.close()
        return
    size = int(s.recv(16).decode().strip())
    received = 0
    with open(save_path, "wb") as f:
        while received < size:
            chunk = s.recv(BUFFER_SIZE)
            if not chunk:
                break
            f.write(chunk)
            received += len(chunk)
            progress_callback(received, size)
    s.close()


if __name__ == "__main__":
    root = TkinterDnD.Tk()
    app = MeshShaeApp(root)
    root.mainloop()
